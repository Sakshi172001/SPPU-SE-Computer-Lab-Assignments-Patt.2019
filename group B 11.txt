
Program : a) Write a Python Program to store roll numbers of student in array who attended training program in random order. 
Write function for searching whether particular student attended training program or not, using Linear search and Sentinel search.
b) Write a Python Program to store roll numbers of student in array who attended training program in sorted order. 
Write function for searching whether particular student attended training program or not, using Binary search and Fibonacci search.

4.Fibonacci Search
5.Exit''')

def linear():
    n=int(input("Enter the count of Students:"))
    print("Enter the %i students roll numbers: "%(n))
    fds=list(map(int,input().split()))
    print("Students Roll Numbers are : ",fds)
    x=int(input("Enter the roll number to be searched: "))
    flag=0
    for i in range(n):
        if(x==fds[i]):
            flag=1
            print("Roll number %i attended training program"%(x))
            break
    if(flag==0):
        print("Roll number %i did not attend training program"%(x))
     
def sentinel():
    n=int(input("Enter the count of Students:"))
    print("Enter the %i students roll numbers: "%(n))
    fds=list(map(int,input().split()))
    print("Students Roll Numbers are : ",fds)
    x=int(input("Enter the roll number to be searched: "))
    last=fds[n-1]
    fds[n-1]=x
    i=0
    while(fds[i]!=x):
        i=i+1
        
    fds[n-1]=last
    if((i<n-1) or (fds[n-1]==x)):
        print("Roll number %i attended training program"%(x))
    else:
        print("Roll number %i did not attend training program"%(x))
        
def Binary():
    n=int(input("Enter the count of Students:"))
    print("Enter the %i students roll numbers in order: "%(n))
    fds=list(map(int,input().split()))
    print("Students Roll Numbers are : ",fds)
    x=int(input("Enter the roll number to be searched: "))
    flag=0
    l=0
    h=n-1
    while(l<=h):
        m=(l+h)//2
        if(x==fds[m]):
            flag=1
            print("Roll number %i attended training program"%(x))
            break
        elif(x<fds[m]):
            h=m-1
        else:
            l=m+1
    if(flag==0):
        print("Roll number %i did not attend training program"%(x))
    
def Fibonacci():
    n=int(input("Enter the count of Students:"))
    print("Enter the %i students roll numbers in order: "%(n))
    fds=list(map(int,input().split()))
    print("Students Roll Numbers are : ",fds)
    x=int(input("Enter the roll number to be searched: "))
    
    m1=1
    m2=0
    m=m1+m2
    while(m<n):
        m2=m1
        m1=m
        m=m1+m2
    flag=0
    offset=-1
    while(m>1):
        i=offset+m2
        if(fds[i]<x):
            m=m1
            m1=m2
            m2=m-m1
            offset=i
        elif(fds[i]>x):
            m=m2
            m1=m1-m
            m2=m-m1
        else:
            flag=1
            print("Roll number %i attended training program"%(x))
            break
    if(flag==0):
        print("Roll number %i did not attend training program"%(x))
       
while True:
    ch = int(input("Enter Your Choice:"))
    if(ch==1):
        linear()
    elif(ch==2):
        sentinel()
    elif(ch==3):
        Binary()
    elif(ch==4):
        Fibonacci()
    elif(ch==5):
        break

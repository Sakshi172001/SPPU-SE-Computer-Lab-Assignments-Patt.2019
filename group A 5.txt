def two():

    str1=input("Enter string : ")

def three():
    print("3. To check whether given string is palindrome or not")
    str2=input("Enter string : ")
    lenstr2=len(str2)
    j=lenstr2-1
    flag=0
    for i in range(int(lenstr2/2+1)):
        if(str2[i]==str2[j]):
            flag=1
        else:
            break
        j=-1
        
    if(flag==1):
        print("It is palindrome")
    else:
        print("It is not palindrome")

def insertionsort():
    arr=[]
    n=int(input("Enter number of students : "))
    for i in range (0,n):
        ele=int(input("Enter Percentage of Student : "))
        arr.append(ele)
    print("Percentage of students : ",arr)
    for i in range(1,n):
        key=arr[i]
        j=i-1
        while j>=0 and key<arr[j]:
            arr[j+1]=arr[j]
            j-=1
        arr[j+1]=key
    print("Sorted array is : ")
    for i in range(n):
        print(arr[i])
    m=len(arr)
    print("Top 5 Percentage are : ")
    for i in range(m-5,m):
        print("%.2f"%arr[i])

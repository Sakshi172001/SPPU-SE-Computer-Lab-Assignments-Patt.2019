
Program : Write a Python program to compute following computation on matrix:
1. Addition of two matrices
2. Subtraction of two matrices
3. Multiplication of two matrices
4. Transpose of a matrix

def print_matrix(matrix):
    for i in range(len(matrix)):
        for j in range(len(matrix[0])):
            print("\t",matrix[i][j],end = " ")
        print("\n")
        
def addsubmult():
    m=int(input("Enter rows of matrix = "));
    n=int(input("Enter columns of matrix = "));
 
    matrix1=[[0 for j in range(0,n)] for i in range(0,m)]   
    matrix2=[[0 for j in range(0,n)] for i in range(0,m)]    
    
    add_matrix=[[0 for j in range(0,n)] for i in range(0,m)] 
    sub_matrix=[[0 for j in range(0,n)] for i in range(0,m)] 
    mult_matrix=[[0 for j in range(0,n)] for i in range(0,m)]
    
    print("Enter first matrix elements : ")
    for i in range(0,m):
        for j in range(0,n):
            matrix1[i][j]= int(input("Enter an element = "))
            
    print("Enter second matrix elements : ")
    for i in range(0,m):
        for j in range(0,n):
            matrix2[i][j]=int(input("Enter an element = "))

    for i in range(0,m):
        for j in range(0,n):
            add_matrix[i][j]=matrix1[i][j]+matrix2[i][j]
            
    for i in range(0,m):
        for j in range(0,n):
            sub_matrix[i][j]=matrix1[i][j]-matrix2[i][j]
            
    for i in range(0,m):
        for j in range(0,n):
            mult_matrix[i][j]=matrix1[i][j]*matrix2[i][j]
 
#printing input matrices
    print(" Matrix 1")
    print_matrix(matrix1)
    print(" Matrix 2")
    print_matrix(matrix2)

# printing resultant matrix
    print("Resultant matrix after Addition : ")
    print_matrix(add_matrix)

    print("Resultant matrix after Subtraction : ")
    print_matrix(sub_matrix)
    
    print("Resultant matrix after Multiplication : ")
    print_matrix(mult_matrix)
     
def trans():

    p=int(input("Enter number of rows: "))
    q=int(input("Enter number of columns: "))
    print("Enter the elements of Matrix")
    matrix=[[int(input()) for i in range(q)] for j in range(p)]
    print("Matrix : ")
    for i in range(p):
        for j in range(q):
            print(format(matrix[i][j],"<4"),end="")
        print()

    print("Transposed matrix : ")    
    result=[[0 for i in range(p)] for j in range(q)]
    for i in range(q):
        for j in range(p):
            result[i][j]=matrix[j][i]
        
    for i in range(q):
        for j in range(p):
            print(format(result[i][j],"<4"),end="")
        print()
